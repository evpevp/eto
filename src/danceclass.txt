		 if dance<100:
		 dance+=RAND(5,15)

		 end
     act 'Уйти':gt'fit','razd'
     end

		 if stren<80:'Учитель объяснил вам что для танцев на шесте, кроме умения танцевать, необходима еще и физическая сила.'
		 end
     act 'Уйти':gt'fit','razd'
     end
